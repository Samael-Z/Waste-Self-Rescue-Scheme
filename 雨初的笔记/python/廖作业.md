  ```python
  def trim(s):
  
      l = len(s)
      if l == 0:
          return ''
      lct = 0
      for i in range(l):
          if s[i] == ' ':
              lct += 1
          else:
              break
  
      t = -1
      rct = -1
      if s[-1] != ' ':
          return s[lct:]
      while (t > (-l)):
          if s[t] == ' ':
              rct -= 1
              t -= 1
          else: break
      ret_t = s[lct:rct + 1]
      return ret_t
  ```
