  ```python
  def trim(s):
  
      l = len(s)
      if l == 0:
          return ''
      lct = 0
      for i in range(l):
          if s[i] == ' ':
              lct += 1
          else:
              break
  
      t = -1
      rct = -1
      if s[-1] != ' ':
          return s[lct:]
      while (t > (-l)):
          if s[t] == ' ':
              rct -= 1
              t -= 1
          else: break
      ret_t = s[lct:rct + 1]
      return ret_t
  ```

+ `map/reduce`

  ```python
  def nor(name):
      l = len(name)
      res = ''
      for i in range(l):
          if i == 0:
              res += name[i].upper()
          else:
              res += name[i].lower()
      return res
  L = ['adam', 'LISA', 'barT']
  L2 = map(nor, L)
  print(list(L2))
  ```

   ```python
  def prod(l):
      L2 = reduce(temp, l)
      return L2
   ```

  ```python
  def is_palindrome(n):
      n1 = str(n)
      l = math.floor(len(n1) / 2)
      for i in range(l):
  ```

+ 返回函数（闭包）

  ```python
  def createCounter():
      b = 0
      def counter():
          nonlocal b
          b += 1
          return b
  
      return counter
  ```
