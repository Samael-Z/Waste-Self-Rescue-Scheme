def gcd(a, b):
    return a if not b else gcd(b, a % b)

def lcm(a, b):

def bs(a):
    l = len(a)
    for i in range(l):
        for j in range(l - i - 1):
            if a[j] > a[j + 1]:
                a[j], a[j + 1] = a[j + 1], a[j]


def cs(a):
    l = len(a)
    for i in range(l):
        m = i
        for j in range(i + 1, l):
            if a[j] < a[m]:
                m = j
        a[m], a[i] = a[i], a[m]
        
        
def partition(a, left, right):
    temp = a[left]
    while left < right:
        while left < right and a[right] > temp:
            right -= 1
        a[left] = a[right]
        while left < right and a[left] <= temp:
            left += 1
        a[right] = a[left]
    a[left] = temp
    return left


def qs(a, left, right):
    if left < right:
        pos = partition(a, left, right)
        qs(a, left, pos - 1)
        qs(a, pos + 1, right)
        
        
        
        
def ms(a, l, r):
    if l >= r:
        return
    mid = (l + r) // 2
    ms(a, l, mid)

def insert_sort(a):
    l = len(a)
    for i in range(1, l):
        t = a[i]
        j = i - 1
        while j >= 0 and a[j] > t:
            a[j + 1] = a[j]
            a[j] = t
