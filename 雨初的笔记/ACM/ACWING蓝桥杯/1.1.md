```cpp
#include <stdio.h>

#include <climits>
#include <iostream>
using namespace std;

int n;

void dfs(int u, int state)
{
    if (u == n)
    {
        for (int i = 0; i < n; i++)
            if (state >> i & 1)
                printf("%d ", i + 1);
        cout << endl;
        return;
    }

    dfs(u + 1, state);  // 不选这一个
    dfs(u + 1, state | (1 << u));
}

int main()
{
    cin >> n;
    dfs(0, 0);
    return 0;
}
```

+ 用位运算来记录当前选的数，或者也可以用数组来存。
