```cpp
#include <stdio.h>
#include <string.h>

#include <algorithm>
#include <iostream>
#include <map>
#include <queue>
#include <stack>
#define xx first
#define yy second
using namespace std;
typedef pair<int, int> pii;
const int N = 30;
int matrix[N][N];
int n, m;
pii dest;
vector<pii> path;
int dx[] = {0, -1, 0, 1}, dy[] = {-1, 0, 1, 0};
int g_f = false;
pii g_start;
void print(int n)
{
    if (g_f == false)
        g_f = true;
    printf("(%d,%d)", g_start.xx, g_start.yy);
    for (auto i : path)
    {
            printf("->(%d,%d)", i.xx, i.yy);
    }
    puts("");
}

bool judge(int x, int y)
{
    if (x <= 0 || y <= 0)
        return false;
    if (x > n || y > m)
        return false;
    if (matrix[x][y] == 0)
        return false;
    return true;
}
void dfs(int n, pii start)
{
    // printf("dfs(%d {%d %d)\n", n, start.xx, start.yy);
    if (start == dest)
    {
        print(n);
        return;
    }
    for (int i = 0; i < 4; i++)
    {
        int tx = start.xx + dx[i];
        int ty = start.yy + dy[i];
        if (judge(tx, ty))
        {
            matrix[tx][ty] = 0;  // 堵墙
            pii t = make_pair(tx, ty);
            path.push_back(t);
            dfs(n + 1, t);
            path.pop_back();
            matrix[tx][ty] = 1;  // 回溯
        }
    }
}

int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            scanf("%d", &matrix[i][j]);
    pii start;
    cin >> start.xx >> start.yy >> dest.xx >> dest.yy;
    matrix[start.xx][start.yy] = 0;
    g_start = start;
    dfs(2, start);
    if (g_f == false)
        cout << -1 << endl;
    return 0;
}
```
